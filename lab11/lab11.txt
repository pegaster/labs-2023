#include <stdio.h>

int IsSeparator(char c) {
    return (c == ' ') || (c == ',') || (c == '\n') || (c == '\t') || (c == EOF);
}

int main() {
    int sym_count = 0;
    int new_chr = 0;
    long long count = 0;
    while (1) {
        int chr = getchar();
        if (!IsSeparator(chr)) {
            count += 1;
            if (chr >= 'A' && chr <= 'Z') {
                new_chr = (chr - 'A' + count + 3) % 26 + 'A';
            } else if (chr >= 'A' && chr <= 'Z') {
                new_chr = (chr - 'a' + count + 3) % 26 + 'a';
            } else {
                new_chr = chr;
            }
        printf("%c", new_chr);    
        } else {
            count = 0;
            printf("%c", chr);
        }
        if (chr == EOF) {
            break;
        }
    }
    return 0;
}
