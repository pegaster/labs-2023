#include <stdio.h>
#include <string.h>

int count_words_increasing_order(char* sentence) {
    int count = 0;
    char* word = strtok(sentence, " ");
    
    while (word != NULL) {
        int len = strlen(word);
        int is_increasing = 1;
        
        for (int i = 0; i < len - 1; i++) {
            if (word[i] >= word[i + 1]) {
                is_increasing = 0;
                break;
            }
        }
        
        if (is_increasing) {
            count++;
        }
        
        word = strtok(NULL, " ");
    }
    
    return count;
}

int main() {
    char input[100];
    printf("Введите строку: ");
    fgets(input, sizeof(input), stdin);
    
    input[strcspn(input, "\n")] = '\0'; // Удаляем символ новой строки
    
    int word_count = count_words_increasing_order(input);
    printf("Количество слов с возрастающим порядком букв: %d\n", word_count);
    
    return 0;
}
