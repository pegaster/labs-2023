#include <stdio.h>
#include <stdbool.h>

bool is_increasing(char* word) {
    int i = 0;
    while (word[i + 1] != '\0') { 
        if (word[i] >= word[i + 1]) {
            return false;
        }
        i++;
    }
    return true;
}

int main() {
    char sentence[100];
    printf("Введите предложение: ");
    fgets(sentence, sizeof(sentence), stdin);

    int word_count = 0;
    char* word = NULL;
    bool in_word = false;

    for (int i = 0; sentence[i] != '\0'; i++) {
        if (sentence[i] != ' ' && sentence[i] != '\n') {
            if (!in_word) {
                in_word = true;
                word = &sentence[i];
            }
        } else {
            if (in_word) {
                in_word = false;
                sentence[i] = '\0';
                if (is_increasing(word)) {
                    word_count++;
                }
            }
        }
    }

    printf("Количество слов с буквами в возрастающем лексикографическом порядке: %d\n", word_count);

    return 0;
}
