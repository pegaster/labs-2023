#!/bin/bash

file="$1"
num_copies="$2"
starting_char="$3"

for ((i=0; i<$num_copies; i++)); do
    next_char=$(printf \\$(printf '%03o' $(( $(printf '%d' "'$starting_char") + $i ))))
    cp "$file" "${file}_$next_char"
done

echo "Создано $num_copies копий файла '$file' с начальным символом '$starting_char'."
