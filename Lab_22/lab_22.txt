\section*{\textbf{Определение интеграла Римана}}\\
\end{center}

\indent{\textbf{а. Разбиения}}\\
\indent{\scshape Определение 1.} \textit{Разбиением P} отрезка [$a$,$b$], $a<b$, называется такая конечная система точек $x_0$,...,$x_n$ этого отрезка, что $a=x_0<x_1<...<x_n=b$.\\
\indentОтрезки [$x_{i-1}, x_i$]($i=1,...,n$) называются \textit{отрезками разбиения P}\\
\indentМаксимум $\lambda$ (P) из длин отрезков разбиения называется \textit{параметром разбиения P}\\
\indent{\scshape Определение 2.} Говорят, что имеется \textit{разбиение} (P, $\xi$) с \textit{отмеченными точками} отрезка $[a, b]$ если имеется разбиение P отрезка $[a, b]$ и в каждом из отрезков [$x_{i-1}, x_i$] разбиения P выбрано по точке $\begin{xi} \end{xi}_{\textit{i}}$ ($i=1,...,n$) \in [$x_{i-1},x_i$] ($i=1,...,n$).\\
\indent Набор $(\begin{xi} \end{xi}_1, ...,\begin{xi} \end{xi}_n)$ обозначается одним символом \begin{xi} \end{xi}.\\
\indent\textbf{b. База в множестве разбиений.} В множестве $\mathscr{P}$ разбиений с отмеченными точками данного отрезка [$a, b$] рассмотрим следующую базу $\mathscr{B}$=$\{B_{\textit{d}}\}$. Элемент $B_{\textit{d}}$, d>0, базы $\mathscr{B}$ есть совокупность всех тех разбиений (P, $\xi$) c отмеченными точками отрезка [$a, b$], для которых $\lambda$(P)< $d$\\
\indent Проверим, что $B_{\textit{d}}$, d>0, - действительно база в $\mathscr{P}$ \\
\indent Во-первых, каким бы ни было число $d$>0, очевидно, существование разбиение P отрезка [$a, b$] с параметром $\lambda$(P)<$d$ (например, разбиение на n конгруэнтных отрезков). Но тогда существует и разбиение (P, $\xi$) с отмеченными точками, для которого $\lambda$(P)<$d$.\\
\indent Во-вторых, если $d_{1}$>0, $d_{2}$>0 и $d$=min\{$d_{1}$, $d_{2}$\}, то, очевидно, $B_{\textit{d}_1} \cap B_{\textit{d}_2} = B_{\textit{d}} \in \mathscr{B}$.\\ 
\indent Итак, $\mathscr{B}$ $=$ \{$B_{\textit{d}}$\} -- действительно база в $\mathscr{P}$.\\
\indent\textbf{c. Интегральная сумма}\\
\indent{\scshape Определение 3.} Если функция $f$ определена на отрезке [$a, b$], а (P, $\xi$) -- разбиение с отмеченными точками этого отрезка, то сумма 
\begin{equation}
\sigma(f; P, \xi):=\sum_{i=1}^n f(\xi_i)\Delta{x_i}
\label{eqation}
\end{equation}
где $\Delta{x_i}=x_i-x_{i-1}$, называется \textit{интегральной суммой} функции $f$, соответствующией разбиению (P, $\xi$) с отмеченными точками отрезка [$a, b$].
\indent Поскольку в \textsc{P} имеется база \textsc{B}, то можно ставить вопрос о пределе по этой базе.\\
\indent\textbf{d. Интеграл Римана.} Пусть $f$ -- функция, заданная на отрезке [$a, b$].\\
\indent{\scshape Определение 4.} Говорят, что число $I$ является \textit{интегралом Римана} от функции $f$ на отрезке [$a, b$], если для любого $\epsilon$>0 найдется число $\delta$>0 такое, что для любого разбиения (P, $\xi$) с отмеченными точками отрезка [$a, b$], параметр которого $\lambda(P)<\delta$, имеет место соотношение $$|I-{\sum_{i=1}^n} f(\xi_i)\Delta{x_i}|<\varepsilon.$$\\
\indent Поскольку разбиения $p = (P, \xi)$, для которых $\lambda$(P)<$d$, составляют элемент $B_{\delta}$ введенной выше базы $\mathscr{B}$ в множестве $\mathscr{P}$ разбиений с отмеченными точками, то определение 4 равносилбно тому, что $$I = \lim_\mathscr{B}f(x),$$ т. е. интеграл $I$ есть предел по базе $\mathscr{B}$ значений интегральных сумм функции $f$, отвечающих разбиению с отмеченными точками отрезка [$a, b$].\\
\indent Базу $\mathscr{B}$ естественно обозначить символом $\lambda(P)\xrightarrow$0, и тогда определение интеграла можно переписать в виде

\begin{equation}
I = \lim_{\lambda(P)\rightarrow0}\sum_{i=1}^nf(\xi_i)\Delta{x_i}
\label{equation}
\end{equation}
\indent Интеграл от функции $f(x)$ по отрезку [$a, b$] обозначается символом 
$$\int_a^bf(x)dx,$$ в котром числа $a, b$ называются \textit{нижним} и \textit{верхним пределом интегрирования} соответственно; $f$ - \textit{подынтегральная функция}, $f(x)dx$ - \textit{подынтегральное выражение}, $x$ - \textit{переменная интегрирования}.\\
\indent Итак, 
\begin{equation}
\boxed{\int_a^bf(x)dx = \lim_{\lambda(P)\rightarrow0}\sum_{i=1}^nf(\xi_i)\Delta{x_i}}
\label{equation}
\end{equation}
\indent{\scshape Определение 5.} Функция $f$ называется \textit{интегрируемой по Риману} на отрезке [$a, b$], если для нее существует указанный в (3) предел интегральных сумм $при \lambda(P)\rightarrow0$ (т. е. если для нее определен интеграл Римана)
